оружие = есть & окрас = тёмный -> тип = военная;
оружие = нету & окрас = светлый -> тип = гражданская;
тип = военная & ракеты = есть & бомбы = нет-> назначение = истребитель;
тип = военная & бомбы = есть & ракеты = нет-> назначение = бомбардировщик;
тип = гражданская & размер = средний -> назначение = пассажирский;
тип = гражданская & размер = большой -> назначение = транспортный;
тип = военная & размер = большой -> преимущество = огневая мощь;
тип = военная & размер = малый -> преимущество = манёвренность;
тип = гражданская & размер = большой -> преимущество = крупные перевозки;
тип = гражданская & размер = малый -> преимущество = индивидуальность;
